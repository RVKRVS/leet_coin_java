class Solution {
    public boolean isPowerOfTwo(int n) {
        int r;
        if (n==1) return true;

        else if(n==0) return false;
        else{
            while(n!=1){
            r=n%2;
            if(r!=0) return false;
            n=n/2;
            
            }
        }
    return true;
    }
}
